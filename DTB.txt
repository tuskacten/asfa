#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[1000],b[2000],x,y,m=2,p=2,dem=1;
    cin>>x>>y;
    a[0]=x; a[1]=y;
    b[0]=x; b[1]=y;
    while(cin>>x>>y)
    {
        if(x==a[m-1])
        {
            a[m]=y;
            m++;
        }
        b[p]=x;
        b[p+1]=y;
        p+=2;
    }
    
    for(int i=0;i<p;i++)
    {
        for(int j=i+1;j<p;j++)
        {
            if(b[j]==b[i])
            {
                for(int k=j;k<p;k++) b[k]=b[k+1];
                p--;
            }
        }
    }
    while(p>1)
    {
        p/=2;
        dem++;
    }
    if(dem==m) cout<<"yes";
    else cout<<"no";
    return 0;
}
